window.nextSlide = (sliderContainer) => {
let currentSlide = sliderContainer.querySelector(".tumbnail_image.active img")
let slides = sliderContainer.querySelectorAll(".tumbnail_image img")
let currentIndex = Array.prototype.slice.call(slides).indexOf(currentSlide) + 1
let mainImage = sliderContainer.querySelector(".card-img img")	

if (currentIndex == slides.length) {
currentIndex = 0
}
currentSlide.parentNode.classList.remove('active')
slides[currentIndex].parentNode.classList.add('active')

mainImage.srcset = slides[currentIndex].srcset
mainImage.src = slides[currentIndex].src.replace('/200/133', '/700/500')
mainImage.alt = slides[currentIndex].alt
currentSlide.parentNode.parentNode.scrollTo({ left: currentSlide.clientWidth * currentIndex, behavior: 'smooth' })


}

currentSlide.parentNode.parentNode.scrollTo({ left: currentSlide.parentNode.clientWidth * currentIndex, behavior: 'smooth' }) scrolls the slider container horizontally to the position of the currentSlide. It calculates the scroll position by multiplying the width of the parent node of the currentSlide by the currentIndex. The scrollTo method -->

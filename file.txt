 window.activateSlide = (sliderContainer, currentSlide) => {
        sliderContainer.querySelector(".thumbnails .thumbnail-image.active").classList.remove('active')
        let slides = sliderContainer.querySelectorAll(".thumbnail-image img")	
        let currentIndex = Array.prototype.slice.call(slides).indexOf(currentSlide)
        let mainImage = sliderContainer.querySelector(".card-img img")
    
        mainImage.alt = currentSlide.alt
        mainImage.srcset = currentSlide.dataset.srcset	
        mainImage.src = currentSlide.dataset.src
        mainImage.width = currentSlide.dataset.width
        mainImage.height = currentSlide.dataset.height
        currentSlide.parentNode.classList.add('active')
    
        currentSlide.parentNode.parentNode.scrollTo({ left: currentSlide.parentNode.clientWidth * currentIndex, behavior: 'smooth' })
    }

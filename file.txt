window.siteSlideshowChangeSlide = (slideshowElement, n) => {  
  let slidesElement = slideshowElement.querySelector('.slider')  
  let item = slidesElement.querySelector('.item')
  
  const is_desktop  = window.matchMedia('(min-width: 1281px)').matches
  const is_tablet   = window.matchMedia('(min-width: 601px) and (max-width: 1280px)').matches
  const is_mobile   = window.matchMedia('(min-width: 320px) and (max-width: 600px)').matches
  
  let viewSlides = JSON.parse(slideshowElement.dataset.viewSlides)
  let showCount;
  let groupSize;
  if (is_desktop) {
    showCount = viewSlides?.desktop
    groupSize = (showCount - (showCount - 1))
  }
  if (is_tablet) {
    showCount = viewSlides?.tablet
    groupSize = (showCount - (showCount - 1))
  }
  if (is_mobile) {
    showCount = viewSlides?.mobile
    groupSize = showCount - 1
  }
  
  let gapSlideSize = parseFloat(window.getComputedStyle(slidesElement,null).getPropertyValue("gap"))  
  let scrollLeft = slidesElement.scrollLeft + (item.clientWidth * groupSize * n) + (n * gapSlideSize)
  let totalWidth = slidesElement.scrollWidth - ((item.clientWidth * (groupSize)) + (gapSlideSize * (groupSize)));
  if (is_mobile) {
    totalWidth = slidesElement.scrollWidth
  }
  
  if((scrollLeft + item.clientWidth + (gapSlideSize * showCount)) >= totalWidth){
    scrollLeft = 0  
  }
  else if((scrollLeft + (item.clientWidth * groupSize)) < 0 ){
    scrollLeft = slidesElement.scrollWidth
  }
    
  const element = document.querySelector('.slider')

  slidesElement.scrollTo({left: scrollLeft, behavior: 'smooth'})
}
